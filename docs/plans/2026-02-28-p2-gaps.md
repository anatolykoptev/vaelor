```go
package codegraph

import "strings"

// symbolComplexity estimates cyclomatic complexity from a function body.
// Heuristic keyword counting (not full AST), fast and language-agnostic.
func symbolComplexity(body string) int {
	if body == "" {
		return 1
	}

	cc := 1
	type kw struct {
		text string
		inc  int
	}
	// Order matters: multi-word before single-word.
	multiWord := []kw{{"else if ", 1}, {"elif ", 1}, {"} else if ", 1}}
	cleaned := body
	for _, k := range multiWord {
		count := strings.Count(cleaned, k.text)
		cc += count * k.inc
		cleaned = strings.ReplaceAll(cleaned, k.text, strings.Repeat(" ", len(k.text)))
	}

	singles := []string{"if ", "for ", "while ", "case ", "catch ", "catch(", "except ", "rescue "}
	for _, k := range singles {
		cc += strings.Count(cleaned, k)
	}

	cc += strings.Count(cleaned, "&&")
	cc += strings.Count(cleaned, "||")
	return cc
}
```

```go
package deadcode

import (
	"path/filepath"
	"sort"
	"strings"

	"github.com/anatolykoptev/go-code/internal/callgraph"
	"github.com/anatolykoptev/go-code/internal/parser"
)

// Options configures dead code detection.
type Options struct {
	IncludeExported bool // include exported (public) symbols — usually false positives
	IncludeTests    bool // include test-file symbols
}

// DeadSymbol is a function/method with zero incoming calls.
type DeadSymbol struct {
	Name       string `json:"name"`
	Kind       string `json:"kind"`
	File       string `json:"file"`
	Package    string `json:"package"`
	StartLine  int    `json:"start_line"`
	Lines      int    `json:"lines"`
	Exported   bool   `json:"exported"`
	Confidence string `json:"confidence"` // high, medium, low
}

// Result is the output of dead code analysis.
type Result struct {
	TotalFunctions int          `json:"total_functions"`
	DeadCount      int          `json:"dead_count"`
	DeadRatio      float64      `json:"dead_ratio"`
	DeadSymbols    []DeadSymbol `json:"dead_symbols"`
}

// entryPointNames are function names that are implicit entry points.
var entryPointNames = map[string]bool{
	"main":     true,
	"init":     true,
	"TestMain": true,
}

// Analyze finds functions/methods with zero incoming calls.
func Analyze(cg *callgraph.CallGraph, opts Options) *Result {
	calledSet := buildCalledSet(cg.Edges)
	result := &Result{}

	for _, sym := range cg.Symbols {
		if sym.Kind != parser.KindFunction && sym.Kind != parser.KindMethod {
			continue
		}
		result.TotalFunctions++

		if calledSet[sym] {
			continue // has callers
		}

		if shouldSkip(sym, opts) {
			continue
		}

		ds := makeDeadSymbol(sym)
		result.DeadSymbols = append(result.DeadSymbols, ds)
	}

	result.DeadCount = len(result.DeadSymbols)
	if result.TotalFunctions > 0 {
		result.DeadRatio = float64(result.DeadCount) / float64(result.TotalFunctions)
	}

	sort.Slice(result.DeadSymbols, func(i, j int) bool {
		return result.DeadSymbols[i].File < result.DeadSymbols[j].File
	})

	return result
}

// buildCalledSet returns the set of symbols that have at least one caller.
func buildCalledSet(edges []callgraph.CallEdge) map[*parser.Symbol]bool {
	called := make(map[*parser.Symbol]bool)
	for _, e := range edges {
		if e.Callee != nil {
			called[e.Callee] = true
		}
	}
	return called
}

// shouldSkip returns true if the symbol should be excluded from dead code.
func shouldSkip(sym *parser.Symbol, opts Options) bool {
	// Always skip entry points.
	if entryPointNames[sym.Name] {
		return true
	}
	// Skip test functions (Test*, Benchmark*, Example*, Fuzz*).
	if isTestFunc(sym.Name) {
		return true
	}
	// Skip test files unless requested.
	if !opts.IncludeTests && isTestFile(sym.File) {
		return true
	}
	// Skip exported symbols unless requested.
	if !opts.IncludeExported && isExported(sym.Name) {
		return true
	}
	return false
}

func makeDeadSymbol(sym *parser.Symbol) DeadSymbol {
	exported := isExported(sym.Name)
	lines := 1
	if sym.EndLine >= sym.StartLine {
		lines = int(sym.EndLine-sym.StartLine) + 1
	}

	confidence := "high"
	if exported {
		confidence = "low" // exported funcs may be called from outside
	} else if sym.Kind == parser.KindMethod {
		confidence = "medium" // methods may satisfy interfaces
	}

	return DeadSymbol{
		Name:       sym.Name,
		Kind:       string(sym.Kind),
		File:       sym.File,
		Package:    filepath.Dir(sym.File),
		StartLine:  int(sym.StartLine),
		Lines:      lines,
		Exported:   exported,
		Confidence: confidence,
	}
}

func isExported(name string) bool {
	if name == "" {
		return false
	}
	return name[0] >= 'A' && name[0] <= 'Z'
}

func isTestFile(path string) bool {
	lower := strings.ToLower(path)
	testSuffixes := []string{"_test.go", "_test.py", ".test.ts", ".test.js", ".spec.ts", ".spec.js"}
	for _, suf := range testSuffixes {
		if strings.HasSuffix(lower, suf) {
			return true
		}
	}
	return strings.Contains(lower, "/test/") || strings.Contains(lower, "/tests/")
}

func isTestFunc(name string) bool {
	prefixes := []string{"Test", "Benchmark", "Example", "Fuzz"}
	for _, p := range prefixes {
		if strings.HasPrefix(name, p) && len(name) > len(p) {
			return true
		}
	}
	return false
}
```

```go
package main

import (
	"context"
	"encoding/json"
	"fmt"
	"log/slog"

	"github.com/anatolykoptev/go-code/internal/analyze"
	"github.com/anatolykoptev/go-code/internal/callgraph"
	"github.com/anatolykoptev/go-code/internal/deadcode"
	"github.com/anatolykoptev/go-code/internal/prompts"
	"github.com/anatolykoptev/go-kit/llm"
	"github.com/modelcontextprotocol/go-sdk/mcp"
)

// DeadCodeInput is the input schema for the dead_code tool.
type DeadCodeInput struct {
	Repo            string `json:"repo" jsonschema_description:"Repository: GitHub slug (owner/repo), full GitHub URL, or absolute local host path"`
	Language        string `json:"language,omitempty" jsonschema_description:"Limit analysis to files of this language (e.g. go, python)"`
	IncludeExported bool   `json:"include_exported,omitempty" jsonschema_description:"Include exported/public functions (usually false positives, default: false)"`
	Focus           string `json:"focus,omitempty" jsonschema_description:"Optional focus area for the LLM narrative (e.g. 'cleanup candidates', 'API surface audit')"`
}

func registerDeadCode(server *mcp.Server, cfg Config, deps analyze.Deps) {
	mcp.AddTool(server, &mcp.Tool{
		Name: "dead_code",
		Description: "Detect dead code (functions/methods never called) in a repository. " +
			"Builds a call graph and identifies uncalled functions with confidence scoring. " +
			"Filters entry points (main, init), test functions, and exported APIs by default.",
	}, func(ctx context.Context, _ *mcp.CallToolRequest, input DeadCodeInput) (*mcp.CallToolResult, any, error) {
		if input.Repo == "" {
			return errResult("repo is required"), nil, nil
		}

		root, cleanup, err := resolveRoot(ctx, input.Repo, input.Language, deps)
		if err != nil {
			return errResult(fmt.Sprintf("resolve repo: %s", err)), nil, nil
		}
		defer cleanup()

		cg, err := callgraph.BuildFromRepo(ctx, root, input.Language, deps.ParseCache)
		if err != nil {
			return errResult(fmt.Sprintf("build call graph: %s", err)), nil, nil
		}

		result := deadcode.Analyze(cg, deadcode.Options{
			IncludeExported: input.IncludeExported,
		})

		data, err := json.MarshalIndent(result, "", "  ")
		if err != nil {
			return errResult(fmt.Sprintf("marshal: %s", err)), nil, nil
		}

		// Optional LLM narrative.
		narrative := ""
		if deps.LLM != nil && result.DeadCount > 0 {
			prompt := fmt.Sprintf("Repo dead code analysis:\n%s", string(data))
			if input.Focus != "" {
				prompt += "\n\nFocus: " + input.Focus
			}
			resp, llmErr := deps.LLM.Complete(ctx, prompts.SystemPromptDeadCode, prompt, llm.WithTemperature(0.2))
			if llmErr != nil {
				slog.Warn("dead_code: LLM narrative failed", slog.Any("error", llmErr))
			} else {
				narrative = resp
			}
		}

		output := map[string]any{
			"analysis":  result,
			"narrative": narrative,
		}
		out, _ := json.MarshalIndent(output, "", "  ")
		return &mcp.CallToolResult{
			Content: []mcp.Content{&mcp.TextContent{Text: string(out)}},
		}, nil, nil
	})
}
```

**Step 2: Create `internal/parser/type_relations.go`**

```go
package parser

import (
	"strings"

	sitter "github.com/smacker/go-tree-sitter"
)

// extractGoParents extracts embedded types from a Go struct_type or interface_type body.
func extractGoParents(typeSpec *sitter.Node, source []byte) []string {
	typeBody := typeSpec.ChildByFieldName("type")
	if typeBody == nil {
		return nil
	}

	var parents []string
	switch typeBody.Type() {
	case "struct_type":
		parents = extractGoStructEmbeds(typeBody, source)
	case "interface_type":
		parents = extractGoInterfaceEmbeds(typeBody, source)
	}
	return parents
}

// extractGoStructEmbeds finds unnamed fields in a struct (embedded types).
func extractGoStructEmbeds(structNode *sitter.Node, source []byte) []string {
	var parents []string
	fieldList := firstChildByType(structNode, "field_declaration_list")
	if fieldList == nil {
		return nil
	}
	for i := 0; i < int(fieldList.NamedChildCount()); i++ {
		field := fieldList.NamedChild(i)
		if field == nil || field.Type() != "field_declaration" {
			continue
		}
		// Embedded field: has type but no name(s).
		nameNode := field.ChildByFieldName("name")
		typeNode := field.ChildByFieldName("type")
		if nameNode == nil && typeNode != nil {
			typeName := extractTypeName(typeNode, source)
			if typeName != "" {
				parents = append(parents, typeName)
			}
		}
	}
	return parents
}

// extractGoInterfaceEmbeds finds embedded interfaces.
func extractGoInterfaceEmbeds(ifaceNode *sitter.Node, source []byte) []string {
	var parents []string
	for i := 0; i < int(ifaceNode.NamedChildCount()); i++ {
		child := ifaceNode.NamedChild(i)
		if child == nil {
			continue
		}
		// In Go's tree-sitter, embedded interfaces appear as type_identifier
		// children of interface_type (not method_spec).
		if child.Type() == "type_identifier" || child.Type() == "qualified_type" {
			parents = append(parents, child.Content(source))
		}
	}
	return parents
}

// extractTypeName extracts a clean type name from a type node,
// stripping pointer (*) and package qualifiers.
func extractTypeName(node *sitter.Node, source []byte) string {
	text := node.Content(source)
	// Strip pointer prefix.
	text = strings.TrimPrefix(text, "*")
	// Use last segment of qualified name (pkg.Type → Type).
	if idx := strings.LastIndexByte(text, '.'); idx >= 0 {
		text = text[idx+1:]
	}
	return text
}

// extractPythonBases extracts base classes from a Python class_definition.
func extractPythonBases(classNode *sitter.Node, source []byte) []string {
	superclasses := classNode.ChildByFieldName("superclasses")
	if superclasses == nil {
		return nil
	}
	var parents []string
	for i := 0; i < int(superclasses.NamedChildCount()); i++ {
		child := superclasses.NamedChild(i)
		if child == nil {
			continue
		}
		name := child.Content(source)
		// Skip common non-inheritance bases.
		if name != "object" && name != "ABC" && name != "metaclass" {
			parents = append(parents, name)
		}
	}
	return parents
}

// extractJavaParents extracts extends/implements from a Java class_declaration.
func extractJavaParents(classNode *sitter.Node, source []byte) []string {
	var parents []string
	superclass := classNode.ChildByFieldName("superclass")
	if superclass != nil {
		// superclass is a "superclass" node containing a type_identifier.
		for i := 0; i < int(superclass.NamedChildCount()); i++ {
			child := superclass.NamedChild(i)
			if child != nil {
				parents = append(parents, child.Content(source))
			}
		}
	}
	interfaces := classNode.ChildByFieldName("interfaces")
	if interfaces != nil {
		// interfaces is a "super_interfaces" node containing a type_list.
		for i := 0; i < int(interfaces.NamedChildCount()); i++ {
			child := interfaces.NamedChild(i)
			if child != nil {
				parents = append(parents, child.Content(source))
			}
		}
	}
	return parents
}

// firstChildByType returns the first named child with the given type.
func firstChildByType(node *sitter.Node, nodeType string) *sitter.Node {
	for i := 0; i < int(node.NamedChildCount()); i++ {
		child := node.NamedChild(i)
		if child != nil && child.Type() == nodeType {
			return child
		}
	}
	return nil
}
```

```go
// buildInheritsEdges creates INHERITS edges from Symbol.Parents.
// For each symbol with Parents, it creates edges to any matching type symbol
// in the same repo. Uses name-only matching (best effort).
func buildInheritsEdges(root string, symbols []*parser.Symbol) []edgeData {
	// Build name → symbol key map for type/struct/interface/class symbols.
	typeKeys := make(map[string]string) // name → "Name:relFile"
	for _, sym := range symbols {
		if sym.Kind == parser.KindStruct || sym.Kind == parser.KindInterface ||
			sym.Kind == parser.KindClass || sym.Kind == parser.KindType {
			relFile := relPath(sym.File, root)
			typeKeys[sym.Name] = sym.Name + ":" + relFile
		}
	}

	var edges []edgeData
	for _, sym := range symbols {
		if len(sym.Parents) == 0 {
			continue
		}
		relFile := relPath(sym.File, root)
		fromKey := sym.Name + ":" + relFile
		for _, parent := range sym.Parents {
			toKey, ok := typeKeys[parent]
			if !ok {
				continue // parent not in this repo
			}
			edges = append(edges, edgeData{
				FromLabel: "Symbol",
				FromKey:   fromKey,
				ToLabel:   "Symbol",
				ToKey:     toKey,
				EdgeLabel: "INHERITS",
				Props:     map[string]string{},
			})
		}
	}
	return edges
}
```

```go
package parser

import (
	"regexp"
	"strings"
)

// fallbackPatterns are regex patterns for common function/class/method declarations.
// Each pattern should have a named group "name" for the symbol name.
var fallbackPatterns = []*fallbackPattern{
	// function/def patterns: def foo(...), func foo(...), fn foo(...), function foo(...)
	{
		re:   regexp.MustCompile(`(?m)^[ \t]*(?:def|func|fn|function|sub)\s+(?P<name>[a-zA-Z_]\w*)\s*\(`),
		kind: KindFunction,
	},
	// class patterns: class Foo, class Foo(Bar), class Foo:
	{
		re:   regexp.MustCompile(`(?m)^[ \t]*class\s+(?P<name>[A-Z]\w*)`),
		kind: KindClass,
	},
	// method patterns (indented function defs inside classes — 2+ spaces/tabs)
	{
		re:   regexp.MustCompile(`(?m)^[ \t]{2,}(?:def|func|fn|function|sub)\s+(?P<name>[a-zA-Z_]\w*)\s*\(`),
		kind: KindMethod,
	},
}

type fallbackPattern struct {
	re   *regexp.Regexp
	kind NodeKind
}

// fallbackParse extracts symbols using regex patterns when no tree-sitter handler exists.
// Returns a best-effort ParseResult. No imports or call extraction.
func fallbackParse(path string, source []byte, lang string) *ParseResult {
	result := &ParseResult{
		File:     path,
		Language: lang,
		Symbols:  make([]*Symbol, 0),
		Imports:  make([]string, 0),
	}

	text := string(source)
	lines := strings.Split(text, "\n")
	seen := make(map[string]struct{})

	for _, fp := range fallbackPatterns {
		nameIdx := fp.re.SubexpIndex("name")
		if nameIdx < 0 {
			continue
		}
		matches := fp.re.FindAllStringSubmatchIndex(text, -1)
		for _, loc := range matches {
			name := text[loc[nameIdx*2]:loc[nameIdx*2+1]]
			dedupeKey := string(fp.kind) + ":" + name
			if _, exists := seen[dedupeKey]; exists {
				continue
			}
			seen[dedupeKey] = struct{}{}

			// Compute line number from byte offset.
			lineNum := countNewlines(text[:loc[0]]) + 1

			// Extract signature (the matched line).
			sig := ""
			if lineNum > 0 && lineNum <= len(lines) {
				sig = strings.TrimSpace(lines[lineNum-1])
			}

			result.Symbols = append(result.Symbols, &Symbol{
				Name:      name,
				Kind:      fp.kind,
				Language:  lang,
				File:      path,
				StartLine: uint32(lineNum),
				EndLine:   uint32(lineNum), // Can't determine end without AST
				Signature: sig,
			})
		}
	}

	return result
}

func countNewlines(s string) int {
	return strings.Count(s, "\n")
}
```

```go
package codegraph

import (
	"github.com/anatolykoptev/go-code/internal/callgraph"
	"github.com/anatolykoptev/go-code/internal/parser"
	"github.com/anatolykoptev/go-code/internal/ranking"
)

const (
	pagerankIterations = 20
	pagerankDamping    = 0.85
)

// computeSymbolPageRank runs PageRank on the CALLS graph and returns scores
// keyed by symbol key ("Name:relFile").
func computeSymbolPageRank(root string, symbols []*parser.Symbol, cg *callgraph.CallGraph) map[string]float64 {
	if len(cg.Edges) == 0 {
		return nil
	}

	// Build directed graph: caller → []callees (using symbol keys).
	graph := make(map[string][]string)
	for _, edge := range cg.Edges {
		if edge.Caller == nil || edge.Callee == nil {
			continue
		}
		callerKey := edge.Caller.Name + ":" + relPath(edge.Caller.File, root)
		calleeKey := edge.Callee.Name + ":" + relPath(edge.Callee.File, root)
		graph[callerKey] = append(graph[callerKey], calleeKey)
	}

	// Ensure all symbols are nodes (even if they have no edges).
	for _, sym := range symbols {
		key := sym.Name + ":" + relPath(sym.File, root)
		if _, ok := graph[key]; !ok {
			graph[key] = nil
		}
	}

	return ranking.PageRank(graph, pagerankIterations, pagerankDamping)
}
```

### Task 6: Incremental Graph Indexing

**Context:** `IndexRepo` currently drops and rebuilds the entire graph when TTL expires. The ingest package already captures `ModTime` per file. We can track mtimes and only re-parse changed files.

**Strategy:** Store per-file mtimes in a separate metadata table. On re-index, compare current mtimes with stored ones. Only re-parse files that changed. Delete old vertices/edges for changed files, insert new ones. This is simpler than full graph diffing.

**Files:**

- Create: `internal/codegraph/incremental.go` — mtime tracking, delta computation
- Create: `internal/codegraph/incremental_test.go` — tests

**Step 1: Create `internal/codegraph/incremental.go`**

```go
package codegraph

import (
	"context"
	"time"

	"github.com/anatolykoptev/go-code/internal/ingest"
)

// fileMtime stores the last-known modification time for a file in a graph.
type fileMtime struct {
	RelPath string
	ModTime time.Time
}

// computeChangedFiles compares current file mtimes with stored ones.
// Returns: files to re-parse (changed or new), paths to delete (removed files).
func computeChangedFiles(current []*ingest.File, stored map[string]time.Time) (changed []*ingest.File, removed []string) {
	currentPaths := make(map[string]bool)
	for _, f := range current {
		currentPaths[f.RelPath] = true
		storedTime, exists := stored[f.RelPath]
		if !exists || !f.ModTime.Equal(storedTime) {
			changed = append(changed, f)
		}
	}
	for path := range stored {
		if !currentPaths[path] {
			removed = append(removed, path)
		}
	}
	return changed, removed
}

// storeFileMtimes saves file mtimes for a graph into the metadata table.
func storeFileMtimes(ctx context.Context, store *Store, repoKey string, files []*ingest.File) error {
	mtimes := make(map[string]time.Time, len(files))
	for _, f := range files {
		mtimes[f.RelPath] = f.ModTime
	}
	return store.UpsertFileMtimes(ctx, repoKey, mtimes)
}

// loadFileMtimes retrieves stored file mtimes for a graph.
func loadFileMtimes(ctx context.Context, store *Store, repoKey string) (map[string]time.Time, error) {
	return store.GetFileMtimes(ctx, repoKey)
}
```

```go
// EnsureFileMtimesTable creates the file mtimes tracking table.
func (s *Store) EnsureFileMtimesTable(ctx context.Context) error {
	_, err := s.pool.Exec(ctx, `CREATE TABLE IF NOT EXISTS code_file_mtimes (
		repo_key TEXT NOT NULL,
		file_path TEXT NOT NULL,
		mod_time TIMESTAMPTZ NOT NULL,
		PRIMARY KEY (repo_key, file_path)
	)`)
	return err
}

// UpsertFileMtimes stores file modification times for a repo.
func (s *Store) UpsertFileMtimes(ctx context.Context, repoKey string, mtimes map[string]time.Time) error {
	// Delete old entries for this repo, then batch insert.
	_, err := s.pool.Exec(ctx, "DELETE FROM code_file_mtimes WHERE repo_key = $1", repoKey)
	if err != nil {
		return err
	}
	for path, mtime := range mtimes {
		_, err = s.pool.Exec(ctx,
			"INSERT INTO code_file_mtimes (repo_key, file_path, mod_time) VALUES ($1, $2, $3)",
			repoKey, path, mtime)
		if err != nil {
			return err
		}
	}
	return nil
}

// GetFileMtimes retrieves stored file modification times for a repo.
func (s *Store) GetFileMtimes(ctx context.Context, repoKey string) (map[string]time.Time, error) {
	rows, err := s.pool.Query(ctx,
		"SELECT file_path, mod_time FROM code_file_mtimes WHERE repo_key = $1", repoKey)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	result := make(map[string]time.Time)
	for rows.Next() {
		var path string
		var modTime time.Time
		if err := rows.Scan(&path, &modTime); err != nil {
			return nil, err
		}
		result[path] = modTime
	}
	return result, rows.Err()
}
```

**Step 5: Modify `IndexRepo` in `index.go` for incremental path**

This is the most complex change. When a cached graph exists but is stale, instead of dropping it entirely:
1. Load stored mtimes
2. Ingest all files (fast — just reads filesystem)
3. Compare mtimes → get changed/removed files
4. If >50% changed, fall back to full rebuild (incremental overhead not worth it)
5. Otherwise: re-parse only changed files, delete old vertices for changed/removed, insert new

Add after the cache check in `IndexRepo`:

```go
// Check if incremental update is possible.
if existing != nil && !isFresh(existing.BuiltAt, existing.TTLSeconds) {
	storedMtimes, mtErr := loadFileMtimes(ctx, store, repoKey)
	if mtErr == nil && len(storedMtimes) > 0 {
		allFiles, ingestErr := ingestFiles(ctx, root)
		if ingestErr == nil {
			changed, removed := computeChangedFiles(allFiles, storedMtimes)
			ratio := float64(len(changed)+len(removed)) / float64(len(allFiles))
			if ratio < 0.5 && len(changed)+len(removed) > 0 {
				// Incremental update possible.
				return incrementalUpdate(ctx, store, root, gname, repoKey, existing,
					allFiles, changed, removed, cfg)
			}
		}
	}
	// Fall through to full rebuild.
	if dropErr := store.DropGraph(ctx, gname, repoKey); dropErr != nil {
		return nil, fmt.Errorf("drop stale graph: %w", dropErr)
	}
}
```

The `incrementalUpdate` function would:
1. Delete graph vertices/edges for changed and removed files
2. Re-parse only the changed files
3. Rebuild call graph for changed files
4. Insert new vertices/edges
5. Update mtimes
6. Update meta with new counts

This is complex — keep it simple in v1: just delete + rebuild the affected file subtrees in the graph using Cypher DELETE queries.
