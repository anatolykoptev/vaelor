```go
// internal/compare/ast_diff.go
package compare

import (
	"fmt"

	sitter "github.com/smacker/go-tree-sitter"
	"github.com/smacker/go-tree-sitter/golang"
	"github.com/smacker/go-tree-sitter/javascript"
	"github.com/smacker/go-tree-sitter/python"
	"github.com/smacker/go-tree-sitter/typescript/typescript"
	"github.com/smacker/go-tree-sitter/rust"
	"github.com/smacker/go-tree-sitter/java"
	"github.com/smacker/go-tree-sitter/c"
	"github.com/smacker/go-tree-sitter/cpp"
	"github.com/smacker/go-tree-sitter/ruby"
	"github.com/smacker/go-tree-sitter/csharp"
	"github.com/smacker/gum"
	"github.com/smacker/gum/tsitter"

	"github.com/anatolykoptev/go-code/internal/parser"
)

// ASTDiffResult holds the structured diff between two symbol bodies.
type ASTDiffResult struct {
	Inserts      int     `json:"inserts"`
	Deletes      int     `json:"deletes"`
	Updates      int     `json:"updates"`
	Moves        int     `json:"moves"`
	TotalActions int     `json:"total_actions"`
	Similarity   float64 `json:"similarity"` // 1.0 = identical, 0.0 = completely different
}

// sitterLanguages maps parser language names to tree-sitter language pointers.
var sitterLanguages = map[string]*sitter.Language{
	"go":         golang.GetLanguage(),
	"python":     python.GetLanguage(),
	"typescript": typescript.GetLanguage(),
	"javascript": javascript.GetLanguage(),
	"rust":       rust.GetLanguage(),
	"java":       java.GetLanguage(),
	"c":          c.GetLanguage(),
	"cpp":        cpp.GetLanguage(),
	"ruby":       ruby.GetLanguage(),
	"csharp":     csharp.GetLanguage(),
}

// DiffSymbolAST computes a structural AST diff between two symbol bodies
// using GumTree (smacker/gum). Both symbols must be the same language.
func DiffSymbolAST(a, b *parser.Symbol) (*ASTDiffResult, error) {
	if a.Language != b.Language {
		return nil, fmt.Errorf("language mismatch: %s vs %s", a.Language, b.Language)
	}

	lang, ok := sitterLanguages[a.Language]
	if !ok {
		return nil, fmt.Errorf("unsupported language for AST diff: %s", a.Language)
	}

	srcA := []byte(a.Body)
	srcB := []byte(b.Body)

	treeA, err := parseSitter(srcA, lang)
	if err != nil {
		return nil, fmt.Errorf("parse A: %w", err)
	}
	treeB, err := parseSitter(srcB, lang)
	if err != nil {
		return nil, fmt.Errorf("parse B: %w", err)
	}

	gumA := tsitter.ToTree(treeA, srcA)
	gumB := tsitter.ToTree(treeB, srcB)

	mappings := gum.Match(gumA, gumB)
	actions := gum.Patch(gumA, gumB, mappings)

	result := &ASTDiffResult{}
	for _, a := range actions {
		switch a.Type {
		case gum.Insert, gum.InsertTree:
			result.Inserts++
		case gum.Delete, gum.DeleteTree:
			result.Deletes++
		case gum.Update:
			result.Updates++
		case gum.Move:
			result.Moves++
		}
	}
	result.TotalActions = result.Inserts + result.Deletes + result.Updates + result.Moves

	// Similarity = ratio of mapped nodes to total nodes.
	nodesA := len(gum.PreOrder(gumA))
	nodesB := len(gum.PreOrder(gumB))
	totalNodes := nodesA + nodesB
	if totalNodes == 0 {
		result.Similarity = 1.0
	} else {
		// Each mapping pairs one node from A with one from B (2 matched nodes per mapping).
		matchedNodes := len(mappings) * 2
		result.Similarity = float64(matchedNodes) / float64(totalNodes)
		if result.Similarity > 1.0 {
			result.Similarity = 1.0
		}
	}

	return result, nil
}

// parseSitter parses source bytes into a tree-sitter root node.
func parseSitter(source []byte, lang *sitter.Language) (*sitter.Node, error) {
	p := sitter.NewParser()
	p.SetLanguage(lang)
	tree, err := p.ParseCtx(nil, nil, source)
	if err != nil {
		return nil, err
	}
	return tree.RootNode(), nil
}
```

Note: The exact `sitter.Parse` API may differ from what `gum/tsitter` expects. Check if `tsitter.ToTree` takes `*sitter.Node` — if it does, pass `tree.RootNode()`. If it takes `*sitter.Tree`, pass `tree` directly. Adjust `parseSitter` accordingly.

```go
// Package impact computes blast radius analysis for symbol changes.
package impact

import (
	"path/filepath"

	"github.com/anatolykoptev/go-code/internal/callgraph"
	"github.com/anatolykoptev/go-code/internal/parser"
)

// thresholds for blast radius classification (from CodeMCP).
const (
	lowMaxCallers    = 5
	lowMaxPackages   = 2
	medMaxCallers    = 20
	medMaxPackages   = 5
)

// Options controls impact analysis.
type Options struct {
	MaxDepth int // max traversal depth (default 5)
}

// AffectedSymbol is a symbol affected by a change, with distance and confidence.
type AffectedSymbol struct {
	Name       string  `json:"name"`
	File       string  `json:"file"`
	Package    string  `json:"package"`
	Distance   int     `json:"distance"`    // hops from target (1 = direct caller)
	Confidence float64 `json:"confidence"`  // 1.0 at distance 1, degrades with depth
}

// Result holds the impact analysis output.
type Result struct {
	Symbol           string            `json:"symbol"`
	Found            bool              `json:"found"`
	DirectCallers    []AffectedSymbol  `json:"direct_callers"`
	TransitiveCallers []AffectedSymbol `json:"transitive_callers"`
	TotalAffected    int               `json:"total_affected"`
	AffectedPackages []string          `json:"affected_packages"`
	BlastRadius      string            `json:"blast_radius"` // none, low, medium, high
	RiskScore        float64           `json:"risk_score"`
}

// Analyze computes the blast radius of changing the named symbol.
func Analyze(cg *callgraph.CallGraph, symbolName string, opts Options) *Result {
	if opts.MaxDepth <= 0 {
		opts.MaxDepth = 5
	}

	result := &Result{Symbol: symbolName}

	// Find target symbol.
	var target *parser.Symbol
	for _, sym := range cg.Symbols {
		if sym.Name == symbolName && (sym.Kind == parser.KindFunction || sym.Kind == parser.KindMethod) {
			target = sym
			break
		}
	}

	if target == nil {
		result.BlastRadius = "none"
		return result
	}
	result.Found = true

	// Build reverse call index: callee → []callers.
	callerIndex := make(map[*parser.Symbol][]callgraph.CallEdge)
	for _, e := range cg.Edges {
		if e.Callee != nil {
			callerIndex[e.Callee] = append(callerIndex[e.Callee], e)
		}
	}

	// BFS to find all transitive callers.
	visited := make(map[*parser.Symbol]bool)
	visited[target] = true

	type queueItem struct {
		sym   *parser.Symbol
		depth int
	}
	queue := []queueItem{{target, 0}}
	pkgSet := make(map[string]bool)

	for len(queue) > 0 {
		item := queue[0]
		queue = queue[1:]

		if item.depth >= opts.MaxDepth {
			continue
		}

		for _, edge := range callerIndex[item.sym] {
			caller := edge.Caller
			if caller == nil || visited[caller] {
				continue
			}
			visited[caller] = true

			distance := item.depth + 1
			confidence := 1.0 - float64(distance-1)*0.2
			if confidence < 0.1 {
				confidence = 0.1
			}

			pkg := filepath.Dir(caller.File)
			pkgSet[pkg] = true

			affected := AffectedSymbol{
				Name:       caller.Name,
				File:       caller.File,
				Package:    pkg,
				Distance:   distance,
				Confidence: confidence,
			}

			if distance == 1 {
				result.DirectCallers = append(result.DirectCallers, affected)
			} else {
				result.TransitiveCallers = append(result.TransitiveCallers, affected)
			}

			queue = append(queue, queueItem{caller, distance})
		}
	}

	result.TotalAffected = len(result.DirectCallers) + len(result.TransitiveCallers)
	for pkg := range pkgSet {
		result.AffectedPackages = append(result.AffectedPackages, pkg)
	}

	result.BlastRadius = classifyBlastRadius(result.TotalAffected, len(result.AffectedPackages))
	result.RiskScore = float64(result.TotalAffected) * (1.0 + float64(len(result.AffectedPackages))*0.1)

	return result
}

// classifyBlastRadius categorizes the blast radius based on caller and package counts.
func classifyBlastRadius(callers, packages int) string {
	if callers == 0 {
		return "none"
	}
	if callers <= lowMaxCallers && packages <= lowMaxPackages {
		return "low"
	}
	if callers <= medMaxCallers && packages <= medMaxPackages {
		return "medium"
	}
	return "high"
}
```

```go
// cmd/go-code/tool_impact.go
package main

import (
	"context"
	"encoding/json"
	"fmt"

	"github.com/anatolykoptev/go-code/internal/analyze"
	"github.com/anatolykoptev/go-code/internal/callgraph"
	"github.com/anatolykoptev/go-code/internal/impact"
	"github.com/anatolykoptev/go-code/internal/prompts"
	"github.com/modelcontextprotocol/go-sdk/mcp"
)

// ImpactInput is the input schema for the impact_analysis tool.
type ImpactInput struct {
	Repo     string `json:"repo" jsonschema_description:"Repository: GitHub slug (owner/repo), full GitHub URL, or absolute local host path"`
	Symbol   string `json:"symbol" jsonschema_description:"Function or method name to analyze impact for"`
	Depth    int    `json:"depth,omitempty" jsonschema_description:"Max traversal depth for transitive callers (default 5, max 10)"`
	Focus    string `json:"focus,omitempty" jsonschema_description:"Subdirectory to limit analysis to"`
	Language string `json:"language,omitempty" jsonschema_description:"Limit to files of this language (e.g. go, python)"`
}

type impactOutput struct {
	*impact.Result
	Narrative string `json:"narrative,omitempty"`
}

func registerImpact(server *mcp.Server, _ Config, deps analyze.Deps) {
	mcp.AddTool(server, &mcp.Tool{
		Name: "impact_analysis",
		Description: "Analyze the blast radius of changing a function or method. " +
			"Shows direct callers, transitive callers, affected packages, " +
			"and risk classification (low/medium/high). " +
			"Useful before refactoring to understand what might break.",
	}, func(ctx context.Context, _ *mcp.CallToolRequest, input ImpactInput) (*mcp.CallToolResult, any, error) {
		return handleImpact(ctx, input, deps)
	})
}

func handleImpact(ctx context.Context, input ImpactInput, deps analyze.Deps) (*mcp.CallToolResult, any, error) {
	if input.Repo == "" {
		return errResult("repo is required"), nil, nil
	}
	if input.Symbol == "" {
		return errResult("symbol is required"), nil, nil
	}

	root, cleanup, err := resolveRoot(ctx, input.Repo, "", deps)
	if err != nil {
		return errResult(fmt.Sprintf("resolve repo: %s", err)), nil, nil
	}
	defer cleanup()

	depth := input.Depth
	if depth <= 0 {
		depth = 5
	}
	if depth > 10 {
		depth = 10
	}

	result, err := callgraph.TraceRepo(ctx, callgraph.TraceRepoInput{
		Root:     root,
		Symbol:   input.Symbol,
		Focus:    input.Focus,
		Language: input.Language,
		Opts: callgraph.TraceOpts{
			Direction: "callers",
			MaxDepth:  depth,
		},
	})
	if err != nil {
		return errResult(fmt.Sprintf("trace: %s", err)), nil, nil
	}

	// Also need the raw CallGraph for impact analysis.
	// Re-parse to get the full call graph (TraceRepo returns TraceResult, not CallGraph).
	// Use callgraph.BuildFromRepo to get the CallGraph directly.
	cg, err := callgraph.BuildFromRepo(ctx, callgraph.TraceRepoInput{
		Root:     root,
		Focus:    input.Focus,
		Language: input.Language,
	})
	if err != nil {
		return errResult(fmt.Sprintf("build call graph: %s", err)), nil, nil
	}

	impactResult := impact.Analyze(cg, input.Symbol, impact.Options{
		MaxDepth: depth,
	})

	output := impactOutput{Result: impactResult}

	// LLM narrative.
	if deps.LLM != nil && impactResult.TotalAffected > 0 {
		resultJSON, _ := json.Marshal(impactResult)
		prompt := fmt.Sprintf("Changed symbol: %s\n\nImpact analysis:\n%s", input.Symbol, string(resultJSON))
		narrative, narErr := deps.LLM.Complete(ctx, prompts.SystemPromptImpact, prompt)
		if narErr == nil {
			output.Narrative = narrative
		}
	}

	// Handle case where TraceRepo found nothing but result is from raw CG.
	_ = result // TraceRepo result used for validation above.

	data, err := json.MarshalIndent(output, "", "  ")
	if err != nil {
		return errResult(fmt.Sprintf("marshal: %s", err)), nil, nil
	}

	return &mcp.CallToolResult{
		Content: []mcp.Content{&mcp.TextContent{Text: string(data)}},
	}, nil, nil
}
```
